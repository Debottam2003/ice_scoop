users Table
----------
create table users(
user_id serial primary key,
email text unique not null,
password text unique not null,
location text,
pin bigint
);

icecreams table
______________
create table icecream(
icecream_id serial primary key,
name text not null,
in_stock boolean not null ,
quantity text,
mfg_date Date,
exp_date Date
);

orders Table
____________
create table orders(
order_id id serial primary key,
user_id integer,
paymentStatus text,
date Date,
time text
Foreign key (user_id) references users(user_id)
);

items Table
___________
create table items(
order_id integer,
icecream_id integer,
quantity integer
Foreign key(order_id) references orders(order_id),
Foreign key(icecream_id) references icecreams(icecream_id)
);

staffs table
____________
create table staffs(
staff_id serial primary key,
email text,
password text,
phone_no integer,
joining Date,
designation text,
salary integer
);
